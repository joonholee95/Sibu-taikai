import numpy as np
import time
In [2]:
size = 10000000 #
In [3]:
def origin():
    start = time.time()
    x = range(size)
    y = range(size)
    z = []
    for i in range(len(x)):
        z.append(x[i] + y[i])
    end = time.time() - start
    return end
In [4]:
def numpy():
    start = time.time()
    x = np.arange(size)
    y = np.arange(size)
    z = x + y
    end = time.time() - start
    return end
In [5]:
t1 = origin()
t2 = numpy()

print("numpy processtime : %.5fs" % t2)
print("numpy is %.2f times faster" % (t1/t2))

from random import randint
import matplotlib.pyplot as plt
pos = 0
walk = [pos]
steps = 1000
for i in range(steps):
    if randint(0, 1): # 1  True, 0  False
        step = 1
    else:
        step = -1
    pos += step
    walk.append(pos)
plt.plot(walk)
plt.title("Random walk with 1 step")
plt.show()
